# Reranker Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Cohere cross-encoder reranking to the search pipeline — the single biggest precision improvement for RAG systems. After hybrid search returns candidates, the reranker re-scores each one by reading query + document together, pushing truly relevant docs to the top.

**Architecture:** New `reranker.ts` module calls the Cohere Rerank API (v2, model `rerank-v3.5`) on the top candidates from hybrid search. The reranker is optional — controlled by a `reranker` field in search props. When enabled, hybrid search fetches more candidates (e.g., 20) and the reranker selects the best N. When disabled (`'none'`), behavior is unchanged. The Cohere API key lives in `.env` alongside the existing OpenAI key.

**Tech Stack:** Cohere Rerank API v2 (fetch-based, no SDK needed), TypeScript, Vitest

// reranker.ts
// Cross-encoder reranking via Cohere Rerank API.
// Takes initial search results and re-scores each one by reading
// query + document content together. Much more accurate than
// embedding similarity alone.

import type { ISearchResultProps } from './ai-search.js';

const COHERE_RERANK_URL = 'https://api.cohere.com/v2/rerank';
const COHERE_RERANK_MODEL = 'rerank-v3.5';

export interface IRerankOptionsProps {
  apiKey: string;
  topN?: number;
  model?: string;
}

interface ICohereRerankResponseResult {
  index: number;
  relevance_score: number;
}

interface ICohereRerankResponse {
  results: ICohereRerankResponseResult[];
}

/**
 * Re-rank search results using Cohere's cross-encoder model.
 *
 * Sends each result's content + the query to Cohere, which scores
 * how well each document answers the query. Results are re-sorted
 * by this relevance score (highest first).
 *
 * Graceful degradation: if the API call fails, returns the original
 * results unchanged. Search should never break because reranking failed.
 */
export async function rerankResults(
  query: string,
  searchResults: ISearchResultProps[],
  options: IRerankOptionsProps,
): Promise<ISearchResultProps[]> {
  if (searchResults.length === 0) return [];

  const topN = options.topN ?? searchResults.length;
  const model = options.model ?? COHERE_RERANK_MODEL;

  const documents = searchResults.map(result => ({
    text: result.content,
  }));

  let response: Response;
  try {
    response = await fetch(COHERE_RERANK_URL, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${options.apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model,
        query,
        documents,
        top_n: topN,
      }),
    });
  } catch (networkError) {
    // Network failure — return originals
    return searchResults;
  }

  if (!response.ok) {
    // API error — return originals (graceful degradation)
    return searchResults;
  }

  const cohereResponse = (await response.json()) as ICohereRerankResponse;

  // Map Cohere results back to our search results, re-sorted by relevance
  const rerankedResults: ISearchResultProps[] = cohereResponse.results.map(
    (cohereResult) => ({
      ...searchResults[cohereResult.index],
      score: cohereResult.relevance_score,
    }),
  );

  return rerankedResults;
}
```

export async function searchHybrid(
  clients: IClientsProps,
  props: IHybridSearchProps,
): Promise<ISearchResultProps[]> {
  const startTime = Date.now();
  const queryEmbedding = await getOrCacheQueryEmbedding(clients, props.query);
  const useReranker = props.reranker === 'cohere' && clients.cohereApiKey;
  const requestLimit = useReranker ? (props.limit ?? 10) * 2 : (props.limit ?? 10);

  const { data, error } = await clients.supabase.rpc('match_documents_hybrid', {
    q_emb: toVectorString(queryEmbedding),
    q_text: props.query,
    p_threshold: props.threshold ?? 0.25,
    p_max_results: requestLimit,
    p_domain: props.domain ?? null,
    p_document_type: props.document_type ?? null,
    p_project: props.project ?? null,
    p_rrf_k: props.reciprocalRankFusionK ?? 60,
  });

  if (error) throw new Error(`Hybrid search failed: ${error.message}`);
  let results = (data ?? []) as ISearchResultProps[];

  if (useReranker && results.length > 0) {
    results = await rerankResults(props.query, results, {
      apiKey: clients.cohereApiKey!,
      topN: props.limit ?? 10,
    });
  }

  logSearchEvaluation(clients.supabase, {
    query: props.query,
    searchMode: useReranker ? 'hybrid+rerank' : 'hybrid',
    results,
    responseTimeMs: Date.now() - startTime,
  });

  return results;
}
```
